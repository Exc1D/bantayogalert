Create `src/app/report/ReportFormDesktopWrapper.tsx` — desktop integration (D-76).

**Purpose:** Render ReportForm inside the WorkspaceDrawer on desktop (>=1280px).

**Implementation pattern (from DesktopShell + WorkspaceDrawer):**
- When this component mounts, it sets `activePanel` to a new panel type 'report-form'
- Import `useUIStore` to call `setActivePanel('report-form')`
- Render `<ReportForm />` which contains the submit button
- On submit success: call `setDrawerOpen(false)` and `setActivePanel(null)`

**Add 'report-form' to ActivePanel type in uiStore:**
Update `src/stores/uiStore.ts`:
```typescript
export type ActivePanel = 'report-detail' | 'contact-detail' | 'announcement-detail' | 'report-form' | 'settings' | null
```

**DesktopWrapper:**
```tsx
import { useEffect } from 'react'
import { useNavigate } from 'react-router-dom'
import { ReportForm } from './ReportForm'
import { useUIStore } from '@/stores/uiStore'
import { useAuth } from '@/lib/auth/AuthProvider'
import { submitReport } from '@/features/report/submitReport'
import { uploadMediaFiles } from '@/features/report/mediaUpload'
import { compressImage } from '@/features/report/mediaUpload'
import { clearDraft } from '@/features/report/useReportDraft'

export function ReportFormDesktopWrapper() {
  const { setActivePanel, setDrawerOpen } = useUIStore()
  const { user } = useAuth()
  const navigate = useNavigate()

  useEffect(() => {
    // Open the drawer with report-form panel
    setActivePanel('report-form')
    return () => {
      setActivePanel(null)
    }
  }, [setActivePanel])

  async function handleFormSubmit(data: ReportFormData) {
    if (!user) return

    // 1. Compress images
    const compressedFiles = data.photos
      ? await Promise.all(data.photos.map((f: File) => compressImage(f)))
      : []

    // 2. Call submitReport CF to get reportId
    const { reportId } = await submitReport({
      type: data.type,
      severity: data.severity,
      description: data.description,
      municipalityCode: data.municipalityCode,
      barangayCode: data.barangayCode,
      exactLocation: { lat: data.location.lat, lng: data.location.lng },
      mediaUrls: [],  // will upload after
    })

    // 3. Upload media files
    if (compressedFiles.length > 0) {
      await uploadMediaFiles(compressedFiles, reportId)
    }

    // 4. Clear draft
    await clearDraft(user.uid)

    // 5. Close drawer and navigate to track page
    setDrawerOpen(false)
    setActivePanel(null)
    navigate(`/app/track/${reportId}`)
  }

  return <ReportForm onSubmit={handleFormSubmit} />
}
```

Note: The actual submit integration with media upload happens in plan 05-03 ReportForm component. Here we just wire the navigation.

2. **Real-time listener on report_private:**
```tsx
import { doc, onSnapshot } from 'firebase/firestore'
import { getFirestore } from 'firebase/firestore'

const [reportPrivate, setReportPrivate] = useState<ReportPrivate | null>(null)

useEffect(() => {
  if (!reportId) return
  const db = getFirestore()
  const ref = doc(db, 'report_private', reportId)
  const unsubscribe = onSnapshot(ref, (snap) => {
    if (snap.exists()) setReportPrivate(snap.data() as ReportPrivate)
  })
  return unsubscribe
}, [reportId])
```
