**uploadMediaFiles:**
```typescript
import { getStorage, ref, uploadBytes, getDownloadURL } from 'firebase/storage'

export async function uploadMediaFiles(
  files: File[],
  reportId: string,
  onProgress?: (fileIndex: number, percent: number) => void
): Promise<string[]> {
  const storage = getStorage()
  const urls: string[] = []

  for (let i = 0; i < files.length; i++) {
    const file = files[i]
    const filename = `${Date.now()}-${i}.jpg`
    const storageRef = ref(storage, `reports/${reportId}/${filename}`)

    const snapshot = await uploadBytes(storageRef, file)
    const url = await getDownloadURL(snapshot.ref)
    urls.push(url)
    onProgress?.(i, 100)
  }

  return urls
}
```

**Validation:**
- Max 5 images (enforced in ReportFormSchema, not here)
- Accept: image/jpeg, image/png, image/webp (enforced in file input accept attribute)
- File input pattern: `<input type="file" accept="image/jpeg,image/png,image/webp" multiple capture="environment">`

```typescript
import { httpsCallable } from 'firebase/functions'
import { getFunctions } from 'firebase/app'
import type { IncidentType, Severity } from '@/types/report'

interface SubmitReportData {
  type: IncidentType
  severity: Severity
  description: string
  municipalityCode: string
  barangayCode: string
  exactLocation: { lat: number; lng: number }
  mediaUrls: string[]
}

export async function submitReport(data: SubmitReportData): Promise<{ reportId: string }> {
  const fn = getFunctions()
  const callable = httpsCallable<SubmitReportData, { reportId: string }>(fn, 'submitReport')
  const result = await callable(data)
  return result.data
}
```

This is the client-side counterpart of the Cloud Function created in plan 05-01. It takes the same data shape and returns `{ reportId }`.
