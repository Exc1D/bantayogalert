```tsx
const [compressedFiles, setCompressedFiles] = useState<File[]>([])
const [fileError, setFileError] = useState<string | null>(null)

async function handleFileChange(e: React.ChangeEvent<HTMLInputElement>) {
  const files = Array.from(e.target.files ?? [])
  if (files.length > 5) {
    setFileError('Maximum 5 photos allowed. Please remove some photos.')
    return
  }
  setFileError(null)
  const compressed = await Promise.all(files.map(f => compressImage(f)))
  setCompressedFiles(compressed)

  // Create blob URLs for preview and propagate to parent (StepReview reads these)
  const blobUrls = compressed.map(f => URL.createObjectURL(f))
  setPreviewUrls(blobUrls)
}
```

**Remove photo handler:**

```tsx
function handleRemovePhoto(index: number) {
  setCompressedFiles(prev => prev.filter((_, i) => i !== index))
  const currentUrls = compressedFiles.map(f => URL.createObjectURL(f))
  const newUrls = currentUrls.filter((_, i) => i !== index)
  setPreviewUrls(newUrls)
  onRemovePhoto(index)
}
```
