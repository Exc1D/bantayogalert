## React 18 Strict Mode Double-Mount Issue
In React 18 Strict Mode, useEffect runs twice. Leaflet's MapContainer will throw if initialized twice. The solution is a ref guard:
```tsx
const mapRef = useRef<L.Map | null>(null)
const containerRef = useRef<HTMLDivElement>(null)
const [mapReady, setMapReady] = useState(false)

useEffect(() => {
  if (!containerRef.current || mapRef.current) return  // Guard: don't re-init
  // ... initialize map
  mapRef.current = map
  setMapReady(true)
}, [])
```

## ShellRouter Pattern
```tsx
// Detects viewport and renders appropriate shell
const [isDesktop, setIsDesktop] = useState(false)

useEffect(() => {
  const mq = window.matchMedia('(min-width: 1280px)')
  setIsDesktop(mq.matches)
  const handler = (e: MediaQueryListEvent) => setIsDesktop(e.matches)
  mq.addEventListener('change', handler)
  return () => mq.removeEventListener('change', handler)
}, [])

return isDesktop ? <DesktopShell /> : <MobileShell />
```
