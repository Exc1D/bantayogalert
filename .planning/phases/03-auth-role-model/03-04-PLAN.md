       ```javascript
       }
       ```

    **Image MIME type validation (already present):**
    ```javascript

      return request.resource.contentType.matches('image/.*');
    }
    ```

    **Allowed MIME types (narrower than 'image/.*'):**
    ```javascript
    function isAllowedImage() {
      return request.resource.contentType in ['image/jpeg', 'image/png', 'image/webp'];
    }
    ```

    **Final storage.rules:**
    ```javascript
    rules_version = '2';
    service firebase.storage {
      match /b/{bucket}/o {
        function isAuthenticated() { return request.auth != null; }
        function isProvincialSuperadmin() { return request.auth.token.role == 'provincial_superadmin'; }
        function isImage() {
          return request.resource.contentType in ['image/jpeg', 'image/png', 'image/webp'];
        }
        function isUnder10MB() { return request.resource.size <= 10 * 1024 * 1024; }

        // User profile images
        match /users/{userId}/{filename} {
          allow read: if isAuthenticated();
          allow write: if isAuthenticated() && request.auth.uid == userId && isImage() && isUnder10MB();
        }

        // Report media
        match /reports/{reportId}/media/{filename} {
          allow read: if isAuthenticated();
          allow write: if isAuthenticated() && isImage() && isUnder10MB();
        }

        // Contact avatars
        match /contacts/{contactId}/avatar/{filename} {
          allow read: if isAuthenticated();
          allow write: if isAuthenticated() && isImage() && isUnder10MB();
        }
      }
    }
    ```
