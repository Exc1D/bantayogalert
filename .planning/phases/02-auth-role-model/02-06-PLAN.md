2. Add a more comprehensive helper that handles different scenarios:

```javascript
// Core municipality access helper - used by all collection rules
// Explicitly rejects null municipality tokens to prevent edge cases
function hasMunicipalityAccess(request, targetMunicipality) {
  return (
    request.auth != null &&
    request.auth.token.municipality != null &&
    // provincial_superadmin bypasses all scoping
    (request.auth.token.role == 'provincial_superadmin' ||
      // municipal_admin must match target municipality
      (request.auth.token.role == 'municipal_admin' &&
        request.auth.token.municipality == targetMunicipality))
  )
}

// Helper for documents with municipality field
function canAccessMunicipality(targetMunicipality) {
  return hasMunicipalityAccess(request, targetMunicipality)
}

// Check if user is authenticated with valid municipality
// Rejects null municipality tokens explicitly
function hasValidMunicipality() {
  return request.auth != null && request.auth.token.municipality != null
}

// Check if user is authenticated
function isAuthenticated() {
  return request.auth != null
}

// Check role
function hasRole(role) {
  return isAuthenticated() && request.auth.token.role == role
}
```

3. Keep existing helpers (isMunicipalityAdmin, isProvincialSuperadmin) as aliases for compatibility

Update reports collection rules in firestore.rules:

```javascript
match /reports/{reportId} {
  // Citizens: can read all public reports (province-wide), create their own
  // municipal_admin: can read/write only their municipality
  // provincial_superadmin: can read/write all
  // NULL MUNICIPALITY TOKENS ARE REJECTED - location is critical for responders

  allow read: if isAuthenticated() && hasValidMunicipality() && (
    // Citizens can read reports (public fields only via client, but rule allows read)
    // Admin fields filtered at application level per SEC-04
    hasRole('citizen')
    || hasMunicipalityAccess(request, resource.data.assignedMunicipality)
  );

  allow create: if isAuthenticated() && hasValidMunicipality() && (
    // Citizens and admins can create reports
    // Must specify their municipality (must match token)
    request.resource.data.submitterUid == request.auth.uid
    && request.resource.data.assignedMunicipality == request.auth.token.municipality
    && hasRole('citizen')
  );

  allow update: if isAuthenticated() && hasValidMunicipality() && (
    hasMunicipalityAccess(request, resource.data.assignedMunicipality)
  );

  // No delete by clients - provincial_superadmin via Cloud Function only
  allow delete: if false;
}
```

Note: SEC-04 (citizens can't see admin fields) is enforced at application level, not Firestore rules.
Firestore rules allow read to authenticated users; client must filter fields.
Location is critical for responders - null municipality tokens are explicitly rejected.

match /announcements/{announcementId} {
  // Citizens: read active announcements visible to their municipality or province-wide
  // municipal_admin: read/write own municipality only
  // provincial_superadmin: read/write all scope levels

  allow read: if isAuthenticated() && (
    hasRole('provincial_superadmin')
    || hasMunicipalityAccess(request, resource.data.targetMunicipalities[0])
    // Citizens can read province-wide or multi_municipality if their muni is included
    || (
      hasRole('citizen')
      && resource.data.active == true
      && (
        resource.data.scope == 'province'
        || (
          resource.data.scope == 'municipality'
          && request.auth.token.municipality in resource.data.targetMunicipalities
        )
        || (
          resource.data.scope == 'multi_municipality'
          && request.auth.token.municipality in resource.data.targetMunicipalities
        )
      )
    )
  );

  allow create: if isAuthenticated() && (
    hasRole('provincial_superadmin')
    || (
      hasRole('municipal_admin')
      && request.resource.data.scope == 'municipality'
      && request.auth.token.municipality == request.resource.data.targetMunicipalities[0]
    )
  );

  allow update: if isAuthenticated() && (
    hasRole('provincial_superadmin')
    || (
      hasRole('municipal_admin')
      && hasMunicipalityAccess(request, resource.data.targetMunicipalities[0])
      // Only creator can update draft announcements
      && resource.data.creatorUid == request.auth.uid
    )
  );

  allow delete: if isAuthenticated() && (
    hasRole('provincial_superadmin')
    || resource.data.creatorUid == request.auth.uid
  );
}
