  severity: Severity

  municipalityCode: string
  barangayCode: string

  createdAt: string
  updatedAt: string
  priority?: 1 | 2 | 3 | 4 | 5
  classification?: string
  version: number
  assignedContactSnapshot?: ContactSnapshot

  <action>
    Create `src/hooks/useAdminQueueListener.ts`.

    The hook:
    1. Accepts `municipalityCode: string | null` — null means "all municipalities" (superadmin)
    2. Uses onSnapshot on report_ops collection, filtered by municipalityCode if provided
    3. Joins each report_ops doc with its corresponding reports/{id} document for display data
    4. Returns AdminQueueReport[] with merged data (report_ops fields + reports fields)
    5. Uses TanStack Query cache to store results (queryKey: ['admin-queue', municipalityCode])

    Pattern from useVerifiedReportsListener.ts:
    ```typescript
    useEffect(() => {
      let q = query(collection(db, 'report_ops'))
      if (municipalityCode) {
        q = query(q, where('municipalityCode', '==', municipalityCode))
      }
      const unsubscribe = onSnapshot(q, async (snapshot) => {
        // For each report_ops doc, fetch corresponding reports/{id} doc
        const adminReports = await Promise.all(
          snapshot.docs.map(async (opsDoc) => {
            const reportDoc = await getDoc(doc(db, 'reports', opsDoc.id))
            return { id: opsDoc.id, ...opsDoc.data(), ...reportDoc.data() }
          })
        )
        queryClient.setQueryData(['admin-queue', municipalityCode], adminReports)
      })
      return unsubscribe
    }, [queryClient, municipalityCode])
    ```

    Use getDoc (not getDocs) inside the snapshot handler for individual doc reads.
    Import: collection, query, where, onSnapshot, doc, getDoc from firebase/firestore
    Import: useQueryClient from @tanstack/react-query
    Import: db from @/lib/firebase/config
    Import: AdminQueueReport type (define locally or import from @/types/report if added)
  </action>
