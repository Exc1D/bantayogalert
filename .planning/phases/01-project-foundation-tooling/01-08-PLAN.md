Create .github/workflows/ci.yml (D-17: CI pipeline):

```yaml
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

env:
  NODE_VERSION: '20.x'

jobs:
  build-and-test:
    name: Build and Test
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up Node.js ${{ env.NODE_VERSION }}
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: TypeScript compilation
        run: npm run build

      - name: Run Vitest tests
        run: npm run test -- --run

      - name: Run ESLint
        run: npm run lint

  emulators-and-smoke:
    name: Firebase Emulators and Smoke Tests
    runs-on: ubuntu-latest
    needs: build-and-test

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up Node.js ${{ env.NODE_VERSION }}
        uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Install Playwright browsers
        run: npx playwright install --with-deps

      - name: Start Firebase Emulators
        run: npm run emulators:ci &
        env:
          CI: true

      - name: Wait for emulators to start
        run: |
          sleep 15
          curl -s http://localhost:9099/.well-known/openid-configuration || true
          curl -s http://localhost:8080/ || true

      - name: Run Playwright smoke tests
        run: npx playwright test --project=chromium
        env:
          CI: true

      - name: Stop emulators
        if: always()
        run: pkill -f "firebase emulators" || true
```

Note: Firebase Emulator Suite requires Java. The GitHub Actions runner already has Java pre-installed.
</action>
