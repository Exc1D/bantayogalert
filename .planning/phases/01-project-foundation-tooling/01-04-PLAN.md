```typescript
import { type ReactNode } from 'react'
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { HelmetProvider } from 'react-helmet-async'

// Create a client for React Query
const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 1000 * 60 * 5, // 5 minutes
      gcTime: 1000 * 60 * 10, // 10 minutes (formerly cacheTime)
      retry: 1,
      refetchOnWindowFocus: false,
    },
  },
})

export interface AppProvidersProps {
  children: ReactNode
}

export function AppProviders({ children }: AppProvidersProps) {
  return (
    <HelmetProvider>
      <QueryClientProvider client={queryClient}>
        {/* Zustand stores will be initialized here in Phase 2 */}
        {/* For now, this is a placeholder that React Query and Helmet work */}
        {children}
      </QueryClientProvider>
    </HelmetProvider>
  )
}
```

Note: Zustand stores are initialized in Phase 2 (Domain Model). The actual store initializations (authStore, reportStore, etc.) come when types are defined.
