    ```typescript
    import * as functions from 'firebase-functions'
    import * as admin from 'firebase-admin'
    import { FieldValue } from 'firebase-admin/firestore'
    import { AnnouncementSchema } from '../../src/types/announcement'
    import { sanitizeAnnouncementInput } from '../security/sanitize'
    import { isSuperadmin, isMunicipalAdmin } from '../auth/claims'

    const CreateAnnouncementSchema = AnnouncementSchema  // uses same schema

    export const createAnnouncement = functions.https.onCall(async (data, context) => {
      if (!context.auth) {
        throw new functions.https.HttpsError('unauthenticated', 'Authentication required')
      }

      const parseResult = CreateAnnouncementSchema.safeParse(data)
      if (!parseResult.success) {
        throw new functions.https.HttpsError('invalid-argument', parseResult.error.message)
      }

      const announcementData = parseResult.data
      const claims = context.auth.token

      // Authorization: superadmin can target any scope; municipal_admin can only target their own
      if (!isSuperadmin(claims)) {
        if (announcementData.targetScope.type === 'province') {
          throw new functions.https.HttpsError('permission-denied', 'Only provincial superadmin can issue province-wide announcements')
        }
        const targetMuni = announcementData.targetScope.type === 'municipality'
          ? announcementData.targetScope.municipalityCodes[0]
          : announcementData.targetScope.municipalityCodes[0]  // first of array
        if (!isMunicipalAdmin(claims, targetMuni)) {
          throw new functions.https.HttpsError('permission-denied', 'Cannot create announcement for this municipality')
        }
      }

      const sanitized = sanitizeAnnouncementInput(announcementData as Record<string, unknown>)

      const db = admin.firestore()
      const announcementRef = db.collection('announcements').doc()
      const now = new Date().toISOString()

      await announcementRef.set({
        id: announcementRef.id,
        ...sanitized,
        status: 'draft',
        createdBy: context.auth.uid,
        createdAt: now,
        updatedAt: now,
      })

      return { success: true, id: announcementRef.id }
    })
    ```

    **functions/src/announcements/publishAnnouncement.ts:**
    ```typescript
    import * as functions from 'firebase-functions'
    import * as admin from 'firebase-admin'
    import { FieldValue } from 'firebase-admin/firestore'
    import { AnnouncementStatus } from '../../src/types/announcement'
    import { isSuperadmin, isMunicipalAdmin } from '../auth/claims'
    import { sendAnnouncementPush } from './sendAnnouncementPush'

    const PublishSchema = z.object({
      announcementId: z.string(),
    })

    export const publishAnnouncement = functions.https.onCall(async (data, context) => {
      if (!context.auth) {
        throw new functions.https.HttpsError('unauthenticated', 'Authentication required')
      }

      const parseResult = PublishSchema.safeParse(data)
      if (!parseResult.success) {
        throw new functions.https.HttpsError('invalid-argument', parseResult.error.message)
      }

      const { announcementId } = parseResult.data
      const claims = context.auth.token

      const db = admin.firestore()
      const announcementRef = db.collection('announcements').doc(announcementId)
      const announcementDoc = await announcementRef.get()

      if (!announcementDoc.exists) {
        throw new functions.https.HttpsError('not-found', 'Announcement not found')
      }

      const announcement = announcementDoc.data()!

      if (announcement.status !== AnnouncementStatus.Draft) {
        throw new functions.https.HttpsError('failed-precondition', 'Only draft announcements can be published')
      }

      // Authorization
      if (!isSuperadmin(claims)) {
        const targetMuni = announcement.targetScope.municipalityCodes[0]
        if (!isMunicipalAdmin(claims, targetMuni)) {
          throw new functions.https.HttpsError('permission-denied', 'Cannot publish this announcement')
        }
      }

      const now = new Date().toISOString()

      // Transition to published
      await announcementRef.update({
        status: AnnouncementStatus.Published,
        publishedAt: now,
        updatedAt: now,
      })

      // Trigger async FCM fanout (fire-and-forget from caller's perspective)
      sendAnnouncementPush(announcementId, announcement.targetScope).catch((err) => {
        console.error('FCM fanout failed for announcement', announcementId, err)
      })

      return { success: true, announcementId }
    })
    ```

    **functions/src/announcements/getAnnouncements.ts:**
    ```typescript
    import * as functions from 'firebase-functions'
    import * as admin from 'firebase-admin'
    import { z } from 'zod'
    import { AnnouncementStatus } from '../../src/types/announcement'

    const GetAnnouncementsSchema = z.object({
      // Optional filter by type
      type: z.enum(['alert', 'advisory', 'update', 'all_clear']).optional(),
      // Cursor for pagination (last doc id)
      cursor: z.string().optional(),
      // Limit (default 20, max 50)
      limit: z.number().min(1).max(50).optional().default(20),
    })

    export const getAnnouncements = functions.https.onCall(async (data, context) => {
      if (!context.auth) {
        throw new functions.https.HttpsError('unauthenticated', 'Authentication required')
      }

      const parseResult = GetAnnouncementsSchema.safeParse(data)
      if (!parseResult.success) {
        throw new functions.https.HttpsError('invalid-argument', parseResult.error.message)
      }

      const { type, cursor, limit } = parseResult.data
      const claims = context.auth.token
      const municipalityCode = claims.token?.municipalityCode as string | null
      const isSuperadmin = claims.token?.role === 'provincial_superadmin'

      const db = admin.firestore()

      // Build query: always filter by published status
      let query: admin.firestore.Query = db
        .collection('announcements')
        .where('status', '==', AnnouncementStatus.Published)

      // Scope filtering for non-superadmin
      if (!isSuperadmin && municipalityCode) {
        // Return province-wide OR user's municipality
        // Use union query: (targetScope.type == 'province') OR (targetScope.municipalityCodes contains user's municipality)
        // Since Firestore doesn't support OR, fetch both and merge client-side
        const provinceSnap = await db
          .collection('announcements')
          .where('status', '==', AnnouncementStatus.Published)
          .where('targetScope.type', '==', 'province')
          .orderBy('publishedAt', 'desc')
          .limit(limit)
          .get()

        const muniSnap = await db
          .collection('announcements')
          .where('status', '==', AnnouncementStatus.Published)
          .where('targetScope.type', '==', 'municipality')
          .where('targetScope.municipalityCodes', 'array-contains', municipalityCode)
          .orderBy('publishedAt', 'desc')
          .limit(limit)
          .get()

        const multiMuniSnap = await db
          .collection('announcements')
          .where('status', '==', AnnouncementStatus.Published)
          .where('targetScope.type', '==', 'multi_municipality')
          .where('targetScope.municipalityCodes', 'array-contains', municipalityCode)
          .orderBy('publishedAt', 'desc')
          .limit(limit)
          .get()

        // Merge and sort by publishedAt desc, dedupe by id
        const allDocs = [...provinceSnap.docs, ...muniSnap.docs, ...multiMuniSnap.docs]
          .sort((a, b) => (b.data().publishedAt > a.data().publishedAt ? 1 : -1))
          .slice(0, limit)

        const announcements = allDocs.map((doc) => ({ id: doc.id, ...doc.data() }))
        return { success: true, announcements, cursor: null }
      } else {
        // Superadmin: return all published
        if (type) {
          query = query.where('type', '==', type)
        }
        query = query.orderBy('publishedAt', 'desc').limit(limit)

        const snap = await query.get()
        const announcements = snap.docs.map((doc) => ({ id: doc.id, ...doc.data() }))
        const lastDoc = snap.docs[snap.docs.length - 1]

        return {
          success: true,
          announcements,
          cursor: lastDoc?.id ?? null,
        }
      }
    })
    ```

    ```typescript
    import * as admin from 'firebase-admin'
    import { FieldValue } from 'firebase-admin/firestore'
    import { MulticastMessage } from 'firebase-admin/messaging'
    import { AnnouncementTargetScope } from '../../src/types/announcement'

    const BATCH_SIZE = 500 // FCM sendEachForMulticast limit

    /**
     * Fan out FCM push to all users in the target scope.
     * Sends to individually stored tokens (not topics) for per-recipient delivery tracking.
     * Must be called after announcement status transitions to published.
     */
    export async function sendAnnouncementPush(
      announcementId: string,
      targetScope: AnnouncementTargetScope
    ): Promise<{ sent: number; successCount: number; failureCount: number }> {
      const db = admin.firestore()
      const messaging = admin.messaging()

      // Get announcement data for push content
      const announcementDoc = await db.collection('announcements').doc(announcementId).get()
      if (!announcementDoc.exists) {
        throw new Error(`Announcement ${announcementId} not found`)
      }
      const announcement = announcementDoc.data()!

      // Determine user query based on target scope
      let userQuery: FirebaseFirestore.Query = db.collection('users')

      if (targetScope.type === 'province') {
        // Province-wide: get all users
        userQuery = userQuery
      } else {
        // Municipality or multi_municipality: filter by municipalityCode in target
        userQuery = userQuery.where('municipalityCode', 'in', targetScope.municipalityCodes)
      }

      const usersSnap = await userQuery.get()

      // Collect tokens per user
      const tokenBatches: string[][] = []
      const userIdBatches: string[][] = []
      let currentTokenBatch: string[] = []
      let currentUserIdBatch: string[] = []

      for (const userDoc of usersSnap.docs) {
        const tokensSnap = await db
          .collection('users').doc(userDoc.id)
          .collection('fcmTokens')
          .where('enabled', '==', true)
          .get()

        for (const tokenDoc of tokensSnap.docs) {
          const token = tokenDoc.data().token
          if (currentTokenBatch.length >= BATCH_SIZE) {
            tokenBatches.push(currentTokenBatch)
            userIdBatches.push(currentUserIdBatch)
            currentTokenBatch = []
            currentUserIdBatch = []
          }
          currentTokenBatch.push(token)
          currentUserIdBatch.push(userDoc.id)
        }
      }

      if (currentTokenBatch.length > 0) {
        tokenBatches.push(currentTokenBatch)
        userIdBatches.push(currentUserIdBatch)
      }

      if (tokenBatches.length === 0) {
        return { sent: 0, successCount: 0, failureCount: 0 }
      }

      // Send in batches
      let totalSuccess = 0
      let totalFailure = 0

      for (let i = 0; i < tokenBatches.length; i++) {
        const tokens = tokenBatches[i]
        const userIds = userIdBatches[i]

        const message: MulticastMessage = {
          tokens,
          notification: {
            title: announcement.title,
            body: announcement.body,
          },
          webpush: {
            notification: {
              icon: '/icons/icon-192.png',
              badge: '/icons/badge-72x72.png',
              click_action: '/app/alerts',
            },
            fcmOptions: {
              link: '/app/alerts',
            },
          },
          data: {
            announcementId,
            type: announcement.type,
            severity: announcement.severity,
          },
        }

        const response = await messaging.sendEachForMulticast(message)

        // Upsert delivery status per recipient in notifications subcollection
        const batch = db.batch()
        response.responses.forEach((res, idx) => {
          const notifRef = db.doc(`announcements/${announcementId}/notifications/${userIds[idx]}`)
          batch.set(notifRef, {
            status: res.success ? 'delivered' : 'failed',
            sentAt: FieldValue.serverTimestamp(),
            error: res.error?.message ?? null,
          }, { merge: true })
        })
        await batch.commit()

        totalSuccess += response.successCount
        totalFailure += response.failureCount
      }

      return {
        sent: tokenBatches.reduce((a, b) => a + b.length, 0),
        successCount: totalSuccess,
        failureCount: totalFailure,
      }
    }
    ```
