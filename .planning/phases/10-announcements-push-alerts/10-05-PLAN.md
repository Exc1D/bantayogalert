  <files>src/app/providers.tsx</files>
  <action>
    Read src/app/providers.tsx first to understand its current structure.

    Add to providers.tsx:

    1. Import the hooks:
    ```typescript
    import { useFcmToken } from '@/hooks/useFcmToken'
    import { useMunicipalityTopics } from '@/hooks/useMunicipalityTopics'
    import { onMessageInApp } from '@/lib/firebase/messaging'
    ```

    2. Create a FCMSetup component that handles FCM initialization:
    ```typescript
    function FCMSetup({ userId, municipalityCode }: { userId: string; municipalityCode: string | null }) {
      // Register FCM token
      useFcmToken({ userId, municipalityCode, enabled: true })

      // Subscribe to municipality topics
      useMunicipalityTopics({ municipalityCode, enabled: true })

      // Set up foreground message handler
      useEffect(() => {
        let unsubscribe: (() => void) | undefined

        async function setup() {
          unsubscribe = await onMessageInApp((payload) => {
            // Show a browser notification or update the announcements query cache
            if (Notification.permission === 'granted') {
              new Notification(payload.notification?.title ?? 'Bantayog Alert', {
                body: payload.notification?.body,
                icon: '/icons/icon-192.png',
                data: payload.data,
              })
            }
          })
        }

        setup()
        return () => {
          unsubscribe?.()
        }
      }, [])

      return null // Invisible component
    }
    ```

    3. In the AuthProvider component's inner render (after user is set), add:
    ```tsx
    {user && (
      <FCMSetup
        userId={user.uid}
        municipalityCode={customClaims?.municipalityCode ?? null}
      />
    )}
    ```

    The FCMSetup component should be rendered inside the AuthProvider's authenticated user block. Read providers.tsx to understand the exact placement — typically inside the `{user ? (...) : (...)}` block.

    Note: If providers.tsx uses a different pattern for AuthProvider, adapt accordingly while keeping the key calls to useFcmToken and useMunicipalityTopics after user authentication.
  </action>
