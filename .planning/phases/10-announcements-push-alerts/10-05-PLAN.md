    Read src/app/providers.tsx first to understand its current structure.

    The project uses sonner for toast notifications (check if installed: npx playwright show-report or grep for "sonner" in package.json). If not installed, add it via: npm install sonner

    Add to providers.tsx:

    1. Import sonner and the hooks:
    ```typescript
    import { Toaster, toast } from 'sonner'
    import { useFcmToken } from '@/hooks/useFcmToken'
    import { useMunicipalityTopics } from '@/hooks/useMunicipalityTopics'
    import { onMessageInApp } from '@/lib/firebase/messaging'
    ```

    2. Add the Toaster component inside the providers:
    ```tsx
    <Toaster position="top-center" richColors closeButton />
    ```

    3. Create a FCMSetup component that handles FCM initialization:
    ```typescript
    function FCMSetup({ userId, municipalityCode }: { userId: string; municipalityCode: string | null }) {
      // Register FCM token
      useFcmToken({ userId, municipalityCode, enabled: true })

      // Subscribe to municipality topics
      useMunicipalityTopics({ municipalityCode, enabled: true })

      // Set up foreground message handler with VISIBLE toast
      useEffect(() => {
        let unsubscribe: (() => void) | undefined

        async function setup() {
          unsubscribe = await onMessageInApp((payload) => {
            // Show VISIBLE toast notification (FIX: was just console.log before)
            const title = payload.notification?.title ?? 'Bantayog Alert'
            const body = payload.notification?.body ?? ''

            toast(title, {
              description: body,
              duration: 8000,
              richColors: true,
              icon: '🔔',
              action: {
                label: 'View',
                onClick: () => {
                  window.location.href = '/app/alerts'
                },
              },
            })

            // Also update the announcements query cache to trigger a refetch
            // This makes the new announcement appear immediately in the AlertsFeed
            queryClient.invalidateQueries({ queryKey: ['announcements'] })
          })
        }

        setup()
        return () => {
          unsubscribe?.()
        }
      }, [])

      return null // Invisible component
    }
    ```

    4. In the AuthProvider component's inner render (after user is set), add:
    ```tsx
    {user && (
      <FCMSetup
        userId={user.uid}
        municipalityCode={customClaims?.municipalityCode ?? null}
      />
    )}
    ```

    Note: If providers.tsx uses a different pattern for AuthProvider, adapt accordingly while keeping the key calls to useFcmToken and useMunicipalityTopics after user authentication. The toast notification is the key fix from the previous version which only logged to console.

    IMPORTANT: If sonner is not installed, add it first:
    ```
    npm install sonner
    ```
