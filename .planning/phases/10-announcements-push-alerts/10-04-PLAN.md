    import { AlertTriangle, AlertCircle, Info, CheckCircle, MapPin } from 'lucide-react'
    import { formatDistanceToNow } from 'date-fns'
    import type { Announcement, AnnouncementType, AnnouncementSeverity } from '@/types/announcement'

    const SEVERITY_COLORS: Record<AnnouncementSeverity, { bg: string; text: string; border: string; dot: string }> = {
      info: { bg: 'bg-blue-50', text: 'text-blue-800', border: 'border-blue-200', dot: 'bg-blue-500' },
      warning: { bg: 'bg-amber-50', text: 'text-amber-800', border: 'border-amber-200', dot: 'bg-amber-500' },
      critical: { bg: 'bg-red-50', text: 'text-red-800', border: 'border-red-200', dot: 'bg-red-500' },
    }

    const TYPE_ICONS: Record<AnnouncementType, React.ElementType> = {
      alert: AlertTriangle,
      advisory: AlertCircle,
      update: Info,
      all_clear: CheckCircle,
    }

    const SEVERITY_LABELS: Record<AnnouncementSeverity, string> = {
      info: 'Info',
      warning: 'Warning',
      critical: 'Critical',
    }

    const TYPE_LABELS: Record<AnnouncementType, string> = {
      alert: 'Alert',
      advisory: 'Advisory',
      update: 'Update',
      all_clear: 'All Clear',
    }

    interface AlertCardProps {
      announcement: Announcement
      onClick?: (announcement: Announcement) => void
    }

    export function AlertCard({ announcement, onClick }: AlertCardProps) {
      const { type, severity, targetScope, publishedAt, createdAt } = announcement
      const colors = SEVERITY_COLORS[severity]
      const TypeIcon = TYPE_ICONS[type]
      const time = publishedAt || createdAt
      const relativeTime = time ? formatDistanceToNow(new Date(time), { addSuffix: true }) : ''

      const scopeLabel =
        targetScope.type === 'province'
          ? 'Province-wide'
          : targetScope.type === 'municipality'
          ? targetScope.municipalityCodes[0]?.toUpperCase() ?? 'Municipality'
          : `${targetScope.municipalityCodes.length} municipalities`

      return (
        <button
          type="button"
          onClick={() => onClick?.(announcement)}
          className={`w-full text-left p-4 rounded-lg border ${colors.bg} ${colors.border} hover:opacity-90 transition-opacity focus:outline-none focus:ring-2 focus:ring-primary-500`}
        >
          <div className="flex items-start gap-3">
            {/* Severity dot */}
            <span className={`mt-1.5 w-2 h-2 rounded-full flex-shrink-0 ${colors.dot}`} />

            {/* Content */}
            <div className="flex-1 min-w-0">
              <div className="flex items-center gap-2 mb-1">
                <TypeIcon className={`w-4 h-4 flex-shrink-0 ${colors.text}`} />
                <span className={`text-xs font-medium ${colors.text}`}>
                  {SEVERITY_LABELS[severity]}
                </span>
                <span className="text-xs text-gray-500">·</span>
                <span className="text-xs text-gray-500">{TYPE_LABELS[type]}</span>
              </div>

              <h3 className={`font-semibold text-base ${colors.text} mb-1 line-clamp-1`}>
                {announcement.title}
              </h3>

              <p className={`text-sm ${colors.text} opacity-80 line-clamp-2 mb-2`}>
                {announcement.body}
              </p>

              <div className="flex items-center gap-3 text-xs text-gray-500">
                <span className="flex items-center gap-1">
                  <MapPin className="w-3 h-3" />
                  {scopeLabel}
                </span>
                <span>{relativeTime}</span>
              </div>
            </div>
          </div>
        </button>
      )
    }
    ```

    Note: Uses lucide-react icons already in the project (check existing imports in ReportFeedCard). Uses date-fns for relative time formatting (check if installed — if not, use a simple "X ago" formatter).

    import { useState } from 'react'
    import { RefreshCw, Bell, BellOff } from 'lucide-react'
    import { useAnnouncements } from '@/hooks/useAnnouncements'
    import { AlertCard } from './AlertCard'
    import type { Announcement } from '@/types/announcement'

    interface AlertsFeedProps {
      onAlertClick?: (announcement: Announcement) => void
    }

    export function AlertsFeed({ onAlertClick }: AlertsFeedProps) {
      const { data: announcements, isLoading, isError, refetch, isFetching } = useAnnouncements()
      const [selectedAlert, setSelectedAlert] = useState<Announcement | null>(null)

      if (isLoading) {
        return (
          <div className="flex items-center justify-center py-12">
            <RefreshCw className="w-6 h-6 animate-spin text-gray-400" />
          </div>
        )
      }

      if (isError) {
        return (
          <div className="flex flex-col items-center gap-3 py-12 text-center">
            <BellOff className="w-8 h-8 text-gray-400" />
            <p className="text-sm text-gray-500">Unable to load alerts. Please try again.</p>
            <button
              onClick={() => refetch()}
              className="text-sm text-primary-600 hover:underline"
            >
              Retry
            </button>
          </div>
        )
      }

      if (!announcements || announcements.length === 0) {
        return (
          <div className="flex flex-col items-center gap-3 py-12 text-center">
            <Bell className="w-8 h-8 text-gray-300" />
            <p className="text-sm font-medium text-gray-700">No alerts yet</p>
            <p className="text-xs text-gray-500 max-w-xs">
              Alerts issued by municipal administrators or the provincial government will appear here.
            </p>
          </div>
        )
      }

      return (
        <div className="flex flex-col gap-3">
          {/* Refresh indicator */}
          {isFetching && (
            <div className="flex items-center justify-center py-2">
              <RefreshCw className="w-4 h-4 animate-spin text-gray-400 text-xs" />
              <span className="ml-1 text-xs text-gray-400">Updating...</span>
            </div>
          )}

          {/* Alert list */}
          {announcements.map((announcement) => (
            <AlertCard
              key={announcement.id}
              announcement={announcement}
              onClick={(alert) => {
                setSelectedAlert(alert)
                onAlertClick?.(alert)
              }}
            />
          ))}
        </div>
      )
    }
    ```

    Note: The `onAlertClick` prop is optional — if provided, clicking a card calls it. The selectedAlert state is managed internally but can be lifted to a parent if a detail modal/bottom sheet is needed in a future plan.

    ```typescript
    import { useState } from 'react'
    import { useNavigate } from 'react-router-dom'
    import { AlertTriangle, AlertCircle, Info, CheckCircle, Loader2 } from 'lucide-react'
    import { getFunctions, httpsCallable } from 'firebase/functions'
    import { useAuth } from '@/lib/auth/AuthContext'
    import {
      AnnouncementType,
      AnnouncementSeverity,
      AnnouncementSchema,
    } from '@/types/announcement'
    import type { AnnouncementType as AType, AnnouncementSeverity as ASeverity } from '@/types/announcement'

    const TYPE_OPTIONS: { value: AType; label: string; icon: React.ElementType }[] = [
      { value: AnnouncementType.Alert, label: 'Alert', icon: AlertTriangle },
      { value: AnnouncementType.Advisory, label: 'Advisory', icon: AlertCircle },
      { value: AnnouncementType.Update, label: 'Update', icon: Info },
      { value: AnnouncementType.AllClear, label: 'All Clear', icon: CheckCircle },
    ]

    const SEVERITY_OPTIONS: { value: ASeverity; label: string; color: string }[] = [
      { value: AnnouncementSeverity.Info, label: 'Info', color: 'bg-blue-100 text-blue-800 border-blue-200' },
      { value: AnnouncementSeverity.Warning, label: 'Warning', color: 'bg-amber-100 text-amber-800 border-amber-200' },
      { value: AnnouncementSeverity.Critical, label: 'Critical', color: 'bg-red-100 text-red-800 border-red-200' },
    ]

    // 12 Camarines Norte municipality codes
    const MUNICIPALITIES = [
      { code: 'bas', name: 'Basud' },
      { code: 'bat', name: 'Bato' },
      { code: 'cams', name: 'Camaligan' },
      { code: 'can', name: 'Canaman' },
      { code: 'cap', name: 'Capalonga' },
      { code: 'daet', name: 'Daet' },
      { code: 'jmo', name: 'Jose Panganiban' },
      { code: 'labo', name: 'Labo' },
      { code: 'mer', name: 'Mercedes' },
      { code: 'san', name: 'San Lorenzo' },
      { code: 'sip', name: 'Sipocot' },
      { code: 'sta', name: 'Sta. Elena' },
      { code: 'vin', name: 'Vinzons' },
    ]

    interface CreateAlertFormProps {
      onSuccess?: (announcementId: string) => void
      onCancel?: () => void
    }

    export function CreateAlertForm({ onSuccess, onCancel }: CreateAlertFormProps) {
      const { user } = useAuth()
      const claims = user?.customClaims
      const isSuperadmin = claims?.role === 'provincial_superadmin'
      const myMunicipality = claims?.municipalityCode as string | null
      const navigate = useNavigate()

      const [title, setTitle] = useState('')
      const [body, setBody] = useState('')
      const [type, setType] = useState<AType>(AnnouncementType.Alert)
      const [severity, setSeverity] = useState<ASeverity>(AnnouncementSeverity.Info)
      const [scopeType, setScopeType] = useState<'municipality' | 'multi_municipality' | 'province'>(
        isSuperadmin ? 'municipality' : 'municipality'
      )
      const [selectedMunicipalities, setSelectedMunicipalities] = useState<string[]>(
        myMunicipality ? [myMunicipality] : []
      )
      const [isSubmitting, setIsSubmitting] = useState(false)
      const [error, setError] = useState<string | null>(null)

      // For municipal admin: pre-filled, disabled
      const effectiveMunicipalities = isSuperadmin ? selectedMunicipalities : (myMunicipality ? [myMunicipality] : [])

      async function createAnnouncement(): Promise<string | null> {
        let targetScope
        if (scopeType === 'province') {
          targetScope = { type: 'province' }
        } else if (scopeType === 'municipality') {
          targetScope = { type: 'municipality', municipalityCodes: effectiveMunicipalities as [string] }
        } else {
          targetScope = { type: 'multi_municipality', municipalityCodes: effectiveMunicipalities }
        }

        // Validate with Zod
        const parseResult = AnnouncementSchema.safeParse({
          title,
          body,
          type,
          severity,
          targetScope,
        })

        if (!parseResult.success) {
          setError(parseResult.error.message)
          return null
        }

        const functions = getFunctions()
        const createFn = httpsCallable(functions, 'createAnnouncement')
        const result = await createFn({
          title,
          body,
          type,
          severity,
          targetScope,
        })

        if (result.data?.success) {
          return result.data.id as string
        }
        setError('Failed to create announcement')
        return null
      }

      async function publishAnnouncement(announcementId: string): Promise<boolean> {
        const functions = getFunctions()
        const publishFn = httpsCallable(functions, 'publishAnnouncement')
        const result = await publishFn({ announcementId })

        if (result.data?.success) {
          return true
        }
        setError('Failed to publish announcement')
        return false
      }

      async function handleSubmit(e: React.FormEvent, action: 'create_draft' | 'publish_now') {
        e.preventDefault()
        setError(null)
        setIsSubmitting(true)

        try {
          const announcementId = await createAnnouncement()
          if (!announcementId) {
            setIsSubmitting(false)
            return
          }

          if (action === 'publish_now') {
            const published = await publishAnnouncement(announcementId)
            setIsSubmitting(false)
            if (published) {
              onSuccess?.(announcementId)
              navigate('/app/alerts')
            }
          } else {
            onSuccess?.(announcementId)
            navigate('/app/alerts')
          }
        } catch (err: unknown) {
          const message = err instanceof Error ? err.message : 'Unknown error'
          setError(message)
          setIsSubmitting(false)
        }
      }

      function toggleMunicipality(code: string) {
        setSelectedMunicipalities((prev) =>
          prev.includes(code) ? prev.filter((c) => c !== code) : [...prev, code]
        )
      }

      return (
        <form onSubmit={(e) => handleSubmit(e, 'publish_now')} className="flex flex-col gap-4">
          {/* Title */}
          <div>
            <label htmlFor="alert-title" className="block text-sm font-medium text-gray-700 mb-1">
              Title <span className="text-red-500">*</span>
            </label>
            <input
              id="alert-title"
              type="text"
              value={title}
              onChange={(e) => setTitle(e.target.value)}
              required
              minLength={3}
              maxLength={200}
              placeholder="e.g., Flooding advisory for Basud area"
              className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-primary-500"
            />
          </div>

          {/* Body */}
          <div>
            <label htmlFor="alert-body" className="block text-sm font-medium text-gray-700 mb-1">
              Message <span className="text-red-500">*</span>
            </label>
            <textarea
              id="alert-body"
              value={body}
              onChange={(e) => setBody(e.target.value)}
              required
              minLength={10}
              maxLength={5000}
              rows={4}
              placeholder="Provide detailed information about this alert..."
              className="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-primary-500 resize-none"
            />
          </div>

          {/* Type */}
          <div>
            <label className="block text-sm font-medium text-gray-700 mb-2">
              Type <span className="text-red-500">*</span>
            </label>
            <div className="grid grid-cols-2 gap-2">
              {TYPE_OPTIONS.map(({ value, label, icon: Icon }) => (
                <button
                  key={value}
                  type="button"
                  onClick={() => setType(value)}
                  className={`flex items-center gap-2 px-3 py-2 rounded-lg border text-sm font-medium transition-colors ${
                    type === value
                      ? 'bg-primary-50 border-primary-300 text-primary-700'
                      : 'bg-white border-gray-200 text-gray-600 hover:border-gray-300'
                  }`}
                >
                  <Icon className="w-4 h-4" />
                  {label}
                </button>
              ))}
            </div>
          </div>

          {/* Severity */}
          <div>
            <label className="block text-sm font-medium text-gray-700 mb-2">
              Severity <span className="text-red-500">*</span>
            </label>
            <div className="flex gap-2">
              {SEVERITY_OPTIONS.map(({ value, label, color }) => (
                <button
                  key={value}
                  type="button"
                  onClick={() => setSeverity(value)}
                  className={`flex-1 px-3 py-2 rounded-lg border text-sm font-medium transition-colors ${
                    severity === value ? color : 'bg-white border-gray-200 text-gray-600 hover:border-gray-300'
                  }`}
                >
                  {label}
                </button>
              ))}
            </div>
          </div>

          {/* Scope — only for superadmin */}
          {isSuperadmin && (
            <div>
              <label className="block text-sm font-medium text-gray-700 mb-2">Target Scope</label>
              <div className="flex gap-2 mb-3">
                <button
                  type="button"
                  onClick={() => { setScopeType('municipality'); setSelectedMunicipalities(myMunicipality ? [myMunicipality] : []) }}
                  className={`flex-1 px-3 py-2 rounded-lg border text-sm font-medium ${
                    scopeType === 'municipality' ? 'bg-primary-50 border-primary-300 text-primary-700' : 'bg-white border-gray-200 text-gray-600'
                  }`}
                >
                  My Municipality
                </button>
                <button
                  type="button"
                  onClick={() => setScopeType('multi_municipality')}
                  className={`flex-1 px-3 py-2 rounded-lg border text-sm font-medium ${
                    scopeType === 'multi_municipality' ? 'bg-primary-50 border-primary-300 text-primary-700' : 'bg-white border-gray-200 text-gray-600'
                  }`}
                >
                  Multiple
                </button>
                <button
                  type="button"
                  onClick={() => setScopeType('province')}
                  className={`flex-1 px-3 py-2 rounded-lg border text-sm font-medium ${
                    scopeType === 'province' ? 'bg-primary-50 border-primary-300 text-primary-700' : 'bg-white border-gray-200 text-gray-600'
                  }`}
                >
                  Province-wide
                </button>
              </div>

              {/* Municipality multi-select */}
              {scopeType === 'multi_municipality' && (
                <div className="grid grid-cols-3 gap-2 p-3 bg-gray-50 rounded-lg">
                  {MUNICIPALITIES.map(({ code, name }) => (
                    <label key={code} className="flex items-center gap-2 text-sm cursor-pointer">
                      <input
                        type="checkbox"
                        checked={selectedMunicipalities.includes(code)}
                        onChange={() => toggleMunicipality(code)}
                        disabled={!selectedMunicipalities.includes(code) && selectedMunicipalities.length >= 12}
                        className="rounded border-gray-300 text-primary-600 focus:ring-primary-500"
                      />
                      {name}
                    </label>
                  ))}
                </div>
              )}

              {scopeType === 'municipality' && myMunicipality && (
                <p className="text-sm text-gray-500 p-2 bg-gray-50 rounded-lg">
                  This announcement will target <strong>{MUNICIPALITIES.find(m => m.code === myMunicipality)?.name}</strong> only.
                </p>
              )}
            </div>
          )}

          {/* Municipal admin scope info */}
          {!isSuperadmin && myMunicipality && (
            <p className="text-sm text-gray-500 p-2 bg-gray-50 rounded-lg">
              This announcement will target <strong>{MUNICIPALITIES.find(m => m.code === myMunicipality)?.name}</strong> only.
            </p>
          )}

          {/* Error */}
          {error && (
            <div className="p-3 bg-red-50 border border-red-200 rounded-lg text-sm text-red-700">
              {error}
            </div>
          )}

          {/* Actions */}
          <div className="flex gap-3 pt-2">
            {onCancel && (
              <button
                type="button"
                onClick={onCancel}
                className="flex-1 px-4 py-2 border border-gray-300 rounded-lg text-sm font-medium text-gray-700 hover:bg-gray-50"
              >
                Cancel
              </button>
            )}
            <button
              type="button"
              onClick={(e) => handleSubmit(e as unknown as React.FormEvent, 'create_draft')}
              disabled={isSubmitting}
              className="flex-1 px-4 py-2 border border-gray-300 rounded-lg text-sm font-medium text-gray-700 hover:bg-gray-50 disabled:opacity-50 disabled:cursor-not-allowed flex items-center justify-center gap-2"
            >
              {isSubmitting ? <Loader2 className="w-4 h-4 animate-spin" /> : null}
              Save as Draft
            </button>
            <button
              type="submit"
              disabled={isSubmitting}
              className="flex-1 px-4 py-2 bg-primary-600 text-white rounded-lg text-sm font-medium hover:bg-primary-700 disabled:opacity-50 disabled:cursor-not-allowed flex items-center justify-center gap-2"
            >
              {isSubmitting ? <Loader2 className="w-4 h-4 animate-spin" /> : null}
              Publish Now
            </button>
          </div>
        </form>
      )
    }
    ```

    FIX: Added "Save as Draft" and "Publish Now" buttons. Both call createAnnouncement first. If "Publish Now" is selected, it then calls publishAnnouncement to immediately publish. If "Save as Draft" is selected, it just creates the draft. The draft is accessible via admin queue (future plan).
