    // Standalone Firebase Cloud Messaging service worker
    // IMPORTANT: This file is served as-is from public/ and CANNOT use Vite env vars.
    // Firebase config is delivered at runtime via IndexedDB (written by messaging.ts).
    // Source: Firebase Web SDK client setup docs

    importScripts('https://www.gstatic.com/firebasejs/10.8.0/firebase-app-compat.js');
    importScripts('https://www.gstatic.com/firebasejs/10.8.0/firebase-messaging-compat.js');

    const DB_NAME = 'fcm-config-db';
    const DB_VERSION = 1;
    const STORE_NAME = 'fcm-config';
    const CONFIG_KEY = 'firebase-config';

    /**
     * Open IndexedDB and retrieve Firebase config.
     * Returns a promise that resolves with config object or null.
     */
    function getFirebaseConfigFromIDB() {
      return new Promise((resolve, reject) => {
        const request = indexedDB.open(DB_NAME, DB_VERSION);

        request.onerror = () => reject(request.error);

        request.onupgradeneeded = (event) => {
          const db = event.target.result;
          if (!db.objectStoreNames.contains(STORE_NAME)) {
            db.createObjectStore(STORE_NAME, { keyPath: 'id' });
          }
        };

        request.onsuccess = (event) => {
          const db = event.target.result;
          const transaction = db.transaction(STORE_NAME, 'readonly');
          const store = transaction.objectStore(STORE_NAME);
          const getRequest = store.get(CONFIG_KEY);

          getRequest.onsuccess = () => {
            resolve(getRequest.result?.config || null);
          };
          getRequest.onerror = () => reject(getRequest.error);
        };
      });
    }

    /**
     * Initialize Firebase app with config from IndexedDB.
     * Retries with exponential backoff if config not available yet.
     */
    async function initializeFirebaseWithConfig(retries = 3, delay = 500) {
      for (let i = 0; i < retries; i++) {
        try {
          const config = await getFirebaseConfigFromIDB();
          if (config) {
            firebase.initializeApp(config);
            return firebase.messaging();
          }
          console.log(`[FCM SW] Config not found, retry ${i + 1}/${retries} in ${delay}ms`);
          await new Promise(r => setTimeout(r, delay));
          delay *= 2;
        } catch (err) {
          console.error(`[FCM SW] Error initializing:`, err);
          if (i === retries - 1) throw err;
          await new Promise(r => setTimeout(r, delay));
          delay *= 2;
        }
      }
      return null;
    }

    // Background message handler — called when push notification received while SW is active
    messaging.onBackgroundMessage((payload) => {
      const notificationTitle = payload.notification?.title || 'Bantayog Alert';
      const notificationOptions = {
        body: payload.notification?.body || '',
        icon: '/icons/icon-192.png',
        badge: '/icons/badge-72x72.png',
        data: payload.data,
        actions: [
          { action: 'view', title: 'View Alert' },
          { action: 'dismiss', title: 'Dismiss' },
        ],
        vibrate: [200, 100, 200],
        tag: payload.data?.announcementId || 'default',
      };

      self.addEventListener('notificationclick', (event) => {
        event.notification.close();
        if (event.action === 'view') {
          event.waitUntil(
            self.clients.matchAll({ type: 'window', includeUncontrolled: true }).then((clientList) => {
              // Focus existing window or open new one
              for (const client of clientList) {
                if (client.url.includes('/app/alerts') && 'focus' in client) {
                  return client.focus();
                }
              }
              return self.clients.openWindow('/app/alerts');
            })
          );
        }
      });

      return self.registration.showNotification(notificationTitle, notificationOptions);
    });

    // Initialize on SW activation
    initializeFirebaseWithConfig().then((messagingInstance) => {
      if (messagingInstance) {
        console.log('[FCM SW] Firebase messaging initialized successfully');
      }
    }).catch((err) => {
      console.error('[FCM SW] Failed to initialize Firebase messaging:', err);
    });
    ```

    CRITICAL FIX: This approach avoids hardcoded placeholder strings. The SW reads Firebase config from IndexedDB at runtime, written by the main app's `setSwConfig()` call in messaging.ts. This means the SW works correctly in both development and production without any build-time variable substitution.

    /**
     * Firebase Cloud Messaging initialization for client-side (foreground) push.
     *
     * Background push handling is done in public/firebase-messaging-sw.js.
     * This module handles foreground message handling, SW registration, and
     * runtime config delivery to the SW via IndexedDB.
     *
     * CRITICAL: The service worker cannot use Vite env vars (public/ is not processed).
     * Instead, Firebase config is written to IndexedDB on app init, and the SW reads it.
     */
    import { initializeApp, type FirebaseApp } from 'firebase/app'
    import { getMessaging, isSupported, onMessage, type Messaging } from 'firebase/messaging'
    import { getToken } from 'firebase/messaging'

    const SW_CONFIG_DB = 'fcm-config-db'
    const SW_CONFIG_STORE = 'fcm-config'
    const SW_CONFIG_KEY = 'firebase-config'

    // Re-use the already-initialized Firebase app from config.ts
    let _messaging: Messaging | null = null
    let _app: FirebaseApp | null = null

    /**
     * Write Firebase config to IndexedDB for the service worker to read at runtime.
     * This avoids hardcoding Firebase config in public/firebase-messaging-sw.js.
     */
    async function writeSwConfigToIDB(): Promise<void> {
      const config = {
        apiKey: import.meta.env.VITE_FIREBASE_API_KEY,
        authDomain: import.meta.env.VITE_FIREBASE_AUTH_DOMAIN,
        projectId: import.meta.env.VITE_FIREBASE_PROJECT_ID,
        storageBucket: import.meta.env.VITE_FIREBASE_STORAGE_BUCKET,
        messagingSenderId: import.meta.env.VITE_FIREBASE_MESSAGING_SENDER_ID,
        appId: import.meta.env.VITE_FIREBASE_APP_ID,
      }

      return new Promise((resolve, reject) => {
        const request = indexedDB.open(SW_CONFIG_DB, 1)

        request.onerror = () => reject(request.error)

        request.onupgradeneeded = (event) => {
          const db = (event.target as IDBOpenDBRequest).result
          if (!db.objectStoreNames.contains(SW_CONFIG_STORE)) {
            db.createObjectStore(SW_CONFIG_STORE, { keyPath: 'id' })
          }
        }

        request.onsuccess = (event) => {
          const db = (event.target as IDBOpenDBRequest).result
          const transaction = db.transaction(SW_CONFIG_STORE, 'readwrite')
          const store = transaction.objectStore(SW_CONFIG_STORE)
          store.put({ id: SW_CONFIG_KEY, config })
          transaction.oncomplete = () => resolve()
          transaction.onerror = () => reject(transaction.error)
        }
      })
    }

    export async function getMessagingInstance(): Promise<Messaging | null> {
      if (_messaging) return _messaging

      const supported = await isSupported()
      if (!supported) {
        console.warn('FCM is not supported in this browser')
        return null
      }

      // Write config to IndexedDB before registering SW
      try {
        await writeSwConfigToIDB()
      } catch (err) {
        console.error('[FCM] Failed to write SW config to IndexedDB:', err)
      }

      // Register the standalone SW (not VitePWA's)
      if ('serviceWorker' in navigator) {
        try {
          const reg = await navigator.serviceWorker.register('/firebase-messaging-sw.js', {
            scope: '/',
          })
          console.log('FCM SW registered:', reg.scope)
        } catch (err) {
          console.error('FCM SW registration failed:', err)
        }
      }

      // Initialize messaging with the main app
      const { app: firebaseApp } = await import('./config')
      _app = firebaseApp
      _messaging = getMessaging(firebaseApp)

      return _messaging
    }

    /**
     * Get the current messaging instance (sync).
     * Must call getMessagingInstance() first to initialize.
     */
    export function useMessaging(): Messaging | null {
      return _messaging
    }

    /**
     * Register a foreground message handler.
     * Call this once at app startup (e.g., in providers.tsx).
     *
     * @param callback - Called when a foreground FCM message is received
     * @returns Unsubscribe function
     */
    export async function onMessageInApp(
      callback: (payload: { notification?: { title?: string; body?: string }; data?: Record<string, string> }) => void
    ): Promise<() => void> {
      const messaging = await getMessagingInstance()
      if (!messaging) return () => {}

      const unsubscribe = onMessage(messaging, (payload) => {
        console.log('Foreground FCM message:', payload)
        callback(payload)
      })

      return unsubscribe
    }

    /**
     * Request FCM token for push notifications.
     * Requires VITE_FCM_VAPID_KEY to be set.
     *
     * @returns FCM registration token or null if permission denied or unavailable
     */
    export async function requestFcmToken(): Promise<string | null> {
      const messaging = await getMessagingInstance()
      if (!messaging) return null

      const vapidKey = import.meta.env.VITE_FCM_VAPID_KEY
      if (!vapidKey) {
        console.warn('VITE_FCM_VAPID_KEY not set — FCM token will not be generated')
        return null
      }

      try {
        const token = await getToken(messaging, { vapidKey })
        return token
      } catch (err) {
        console.error('FCM token generation failed:', err)
        return null
      }
    }

    /**
     * Set up FCM service worker config in IndexedDB.
     * Call this early in app startup before registering SW.
     */
    export async function setSwConfig(): Promise<void> {
      try {
        await writeSwConfigToIDB()
      } catch (err) {
        console.error('[FCM] Failed to set SW config:', err)
      }
    }
    ```

    Note: The messaging instance is initialized lazily via getMessagingInstance(). The import from config.ts is done dynamically to avoid circular dependencies.

    /**
     * Hook to get and persist the FCM registration token for the current user.
     * Stores tokens in Firestore subcollection: users/{uid}/fcmTokens/{tokenId}
     * This enables server-side per-recipient delivery tracking (ALR-06).
     *
     * Call this hook inside an AuthContext useEffect after user login.
     */
    import { useEffect, useRef } from 'react'
    import { addDoc, collection, serverTimestamp } from 'firebase/firestore'
    import { db } from '@/lib/firebase/config'
    import { requestFcmToken } from '@/lib/firebase/messaging'

    interface UseFcmTokenOptions {
      userId: string
      municipalityCode: string | null
      /** If false, skips token registration (e.g., user denied notification permission) */
      enabled?: boolean
    }

    export function useFcmToken({ userId, municipalityCode, enabled = true }: UseFcmTokenOptions) {
      const savedRef = useRef(false)

      useEffect(() => {
        if (!enabled || !userId || savedRef.current) return
        savedRef.current = true

        async function saveToken() {
          const token = await requestFcmToken()
          if (!token) return

          try {
            await addDoc(collection(db, 'users', userId, 'fcmTokens'), {
              token,
              municipalityCode,
              enabled: true,
              createdAt: serverTimestamp(),
            })
            console.log('FCM token saved for user:', userId)
          } catch (err) {
            console.error('Failed to save FCM token:', err)
          }
        }

        saveToken()
      }, [userId, municipalityCode, enabled])
    }
    ```

    Note: The token is saved with municipalityCode so server-side queries can filter by municipality without joining to the parent user document. The enabled flag allows callers to skip registration if the user has denied notification permission (can be tracked via a separate preferences flag).

    /**
     * Hook to subscribe the current user to FCM municipality topics.
     *
     * Subscribes to:
     * - province_CMN (all province users)
     * - municipality_{code} (for their specific municipality)
     *
     * Topic subscription is done on login based on user's municipalityCode custom claim.
     * Topics enable the app to receive targeted push without needing per-user token storage.
     * (Token storage is still needed for delivery tracking per ALR-06.)
     *
     * Note: This uses the legacy Firebase topic API via getToken + subscribeToTopic.
     * If the browser does not support this, the function gracefully skips.
     */
    import { useEffect, useRef } from 'react'
    import { requestFcmToken, getMessagingInstance } from '@/lib/firebase/messaging'

    const PROVINCE_TOPIC = 'province_CMN'
    const MUNICIPALITY_TOPIC_PREFIX = 'municipality_'

    interface UseMunicipalityTopicsOptions {
      municipalityCode: string | null
      /** If false, skips topic subscription */
      enabled?: boolean
    }

    /**
     * Subscribe to FCM topics based on municipality.
     * This is a fire-and-forget operation — errors are logged but do not throw.
     */
    export async function subscribeToMunicipalityTopics(
      municipalityCode: string | null
    ): Promise<void> {
      try {
        const token = await requestFcmToken()
        if (!token) {
          console.warn('No FCM token available for topic subscription')
          return
        }

        const messaging = await getMessagingInstance()
        if (!messaging) return

        // Dynamic import to avoid circular dependency
        const { subscribeToTopic } = await import('firebase/messaging')

        // Subscribe to province-wide topic (all users)
        await subscribeToTopic(messaging, token, PROVINCE_TOPIC)

        // Subscribe to municipality topic if user has one
        if (municipalityCode) {
          await subscribeToTopic(messaging, token, `${MUNICIPALITY_TOPIC_PREFIX}${municipalityCode}`)
        }

        console.log('Subscribed to FCM topics for municipality:', municipalityCode)
      } catch (err) {
        console.error('FCM topic subscription failed:', err)
      }
    }

    /**
     * Wrapper hook for useEffect-based subscription.
     * Resubscribes if municipalityCode changes.
     */
    export function useMunicipalityTopics({
      municipalityCode,
      enabled = true,
    }: UseMunicipalityTopicsOptions) {
      const prevMuniRef = useRef<string | null>(null)

      useEffect(() => {
        if (!enabled) return
        if (prevMuniRef.current === municipalityCode) return  // no change
        prevMuniRef.current = municipalityCode

        subscribeToMunicipalityTopics(municipalityCode)
      }, [municipalityCode, enabled])
    }

    /**
     * TanStack Query hook for fetching published announcements.
     *
     * For citizens and municipal admins: returns only announcements targeting
     * their municipality or province-wide (ALR-07).
     *
     * For provincial superadmin: returns all published announcements.
     *
     * Queries Firestore directly (not via callable) since this is a read-only
     * client-side operation with security enforced by Firestore rules.
     */
    import { useQuery } from '@tanstack/react-query'
    import {
      collection,
      query,
      where,
      orderBy,
      limit,
      getDocs,
    } from 'firebase/firestore'
    import { db } from '@/lib/firebase/config'
    import { useAuth } from '@/lib/auth/AuthContext'
    import { AnnouncementStatus } from '@/types/announcement'
    import type { Announcement } from '@/types/announcement'

    export function useAnnouncements(options?: { pageSize?: number }) {
      const { pageSize = 20 } = options ?? {}
      const { user } = useAuth()
      const claims = user?.customClaims

      const isSuperadmin = claims?.role === 'provincial_superadmin'
      const municipalityCode = claims?.municipalityCode as string | null

      return useQuery({
        queryKey: ['announcements', municipalityCode, isSuperadmin],
        queryFn: async (): Promise<Announcement[]> => {
          if (!municipalityCode && !isSuperadmin) return []

          if (isSuperadmin) {
            // Superadmin: all published announcements
            const q = query(
              collection(db, 'announcements'),
              where('status', '==', AnnouncementStatus.Published),
              orderBy('publishedAt', 'desc'),
              limit(pageSize)
            )
            const snap = await getDocs(q)
            return snap.docs.map((doc) => ({ id: doc.id, ...doc.data() } as Announcement))
          }

          // Non-superadmin: province-wide + own municipality
          // Fetch all three in parallel and merge
          const [provinceSnap, muniSnap, multiMuniSnap] = await Promise.all([
            getDocs(query(
              collection(db, 'announcements'),
              where('status', '==', AnnouncementStatus.Published),
              where('targetScope.type', '==', 'province'),
              orderBy('publishedAt', 'desc'),
              limit(pageSize)
            )),
            getDocs(query(
              collection(db, 'announcements'),
              where('status', '==', AnnouncementStatus.Published),
              where('targetScope.type', '==', 'municipality'),
              where('targetScope.municipalityCodes', 'array-contains', municipalityCode!),
              orderBy('publishedAt', 'desc'),
              limit(pageSize)
            )),
            getDocs(query(
              collection(db, 'announcements'),
              where('status', '==', AnnouncementStatus.Published),
              where('targetScope.type', '==', 'multi_municipality'),
              where('targetScope.municipalityCodes', 'array-contains', municipalityCode!),
              orderBy('publishedAt', 'desc'),
              limit(pageSize)
            )),
          ])

          // Merge and dedupe by id
          const allDocs = [
            ...provinceSnap.docs,
            ...muniSnap.docs,
            ...multiMuniSnap.docs,
          ]

          const seen = new Set<string>()
          const announcements: Announcement[] = []
          for (const doc of allDocs) {
            if (!seen.has(doc.id)) {
              seen.add(doc.id)
              announcements.push({ id: doc.id, ...doc.data() } as Announcement)
            }
          }

          // Sort by publishedAt desc and limit
          announcements.sort((a, b) => {
            const aTime = a.publishedAt ?? a.createdAt
            const bTime = b.publishedAt ?? b.createdAt
            return bTime > aTime ? 1 : -1
          })

          return announcements.slice(0, pageSize)
        },
        enabled: true, // Always enabled; empty result for unauthenticated is fine
        staleTime: 1000 * 60 * 2, // 2 minutes — announcements don't change frequently
      })
    }
    ```

    Note: This uses direct Firestore reads since getAnnouncements CF returns the same data.
    If the Firestore security rules don't allow client-side reads (rules allow authenticated read),
    this is the correct approach. The CF getAnnouncements can be used instead if needed.
