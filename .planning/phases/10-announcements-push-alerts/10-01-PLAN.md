  <name>Task 1: Extend Announcement type and Zod schema for multi-municipality support</name>
  <files>src/types/announcement.ts</files>
  <action>
    Replace the Announcement interface and AnnouncementSchema in src/types/announcement.ts with the extended versions below.

    **AnnouncementTargetScope discriminated union (add as new interface before Announcement):**
    ```typescript
    export type AnnouncementTargetScope =
      | { type: 'province' }
      | { type: 'municipality'; municipalityCodes: [string] }  // single municipality
      | { type: 'multi_municipality'; municipalityCodes: string[] }  // 2+ municipalities
    ```

    **Announcement interface — replace targetScope field:**
    ```typescript
    export interface Announcement {
      id: string
      title: string
      body: string
      type: AnnouncementType
      severity: AnnouncementSeverity
      targetScope: AnnouncementTargetScope
      status: AnnouncementStatus
      publishedAt?: string
      cancelledAt?: string
      createdBy: string
      createdAt: string
      updatedAt: string
    }
    ```

    **AnnouncementSchema — replace targetScope z.discriminatedUnion:**
    ```typescript
    export const AnnouncementSchema = z.object({
      title: z.string().min(3).max(200),
      body: z.string().min(10).max(5000),
      type: z.nativeEnum(AnnouncementType),
      severity: z.nativeEnum(AnnouncementSeverity),
      targetScope: z.discriminatedUnion('type', [
        z.object({ type: z.literal('province') }),
        z.object({
          type: z.literal('municipality'),
          municipalityCodes: z.array(z.string()).min(1).max(1),  // exactly 1
        }),
        z.object({
          type: z.literal('multi_municipality'),
          municipalityCodes: z.array(z.string()).min(2).max(12),  // 2-12
        }),
      ]),
    })
    ```

    Keep all existing exports: AnnouncementType, AnnouncementSeverity, AnnouncementStatus, Announcement, AnnouncementSchema, AnnouncementTargetScope.
  </action>
  <verify>
    <automated>grep -c "multi_municipality" src/types/announcement.ts && grep -c "municipalityCodes" src/types/announcement.ts</automated>
  </verify>

<task type="auto">
  <name>Task 2: Update Firestore rules for multi-municipality announcement scope</name>
  <files>firestore.rules</files>
  <action>
    Update the announcements rules in firestore.rules (lines 95-103) to handle the extended targetScope.

    **Replace the existing announcements match block (lines 95-103) with:**
    ```
    // Announcements — read by authenticated, write by admin
    match /announcements/{announcementId} {
      allow read: if isAuthenticated();
      allow create: if isProvincialSuperadmin() ||
        (isMunicipalAdmin(request.resource.data.targetScope.municipalityCodes[0]));
      allow update: if isProvincialSuperadmin() ||
        (resource.data.status == 'draft' &&
         isMunicipalAdmin(request.resource.data.targetScope.municipalityCodes[0]));
      allow delete: if isProvincialSuperadmin();
    }
    ```

    Note: Firestore array access via [0] gets first element. For multi_municipality, provincial_superadmin is required. For single municipality, municipal_admin of that municipality is required. Published announcements cannot be modified (status check on update).
  </action>
  <verify>
    <automated>grep -A5 "match /announcements" firestore.rules | grep -c "municipalityCodes"</automated>
  </verify>
  <done>Firestore rules enforce municipal admin can only create announcements targeting their own municipality; superadmin can target any scope; published announcements are immutable</done>
</task>
